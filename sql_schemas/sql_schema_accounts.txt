BEGIN TRANSACTION;
DROP TABLE IF EXISTS "users";

CREATE TABLE IF NOT EXISTS "users" (
	"id"	INTEGER NOT NULL,
	"username"	INTEGER NOT NULL UNIQUE,
	"email"	TEXT NOT NULL UNIQUE,
	"token"	TEXT NOT NULL,
	"num_login"	INTEGER NOT NULL DEFAULT 0,
	"last_login"	TEXT,
	PRIMARY KEY("id" AUTOINCREMENT)
);

COMMIT;
